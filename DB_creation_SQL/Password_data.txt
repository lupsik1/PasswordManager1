-- Table: public.accounts

-- DROP TABLE public.accounts;

CREATE TABLE public.accounts
(
    password text COLLATE pg_catalog."default" NOT NULL,
    user_email text COLLATE pg_catalog."default" NOT NULL,
    username text COLLATE pg_catalog."default" NOT NULL,
    url text COLLATE pg_catalog."default" NOT NULL,
    appname text COLLATE pg_catalog."default" NOT NULL,
    id integer NOT NULL DEFAULT nextval('accounts_id_seq'::regclass),
    CONSTRAINT accounts_pkey PRIMARY KEY (id),
    CONSTRAINT accounts_check CHECK (user_email ~~ '%@%'::text AND url ~~ 'http%'::text)
)

TABLESPACE pg_default;

ALTER TABLE public.accounts
    OWNER to postgres;

GRANT DELETE, INSERT, SELECT, UPDATE ON TABLE public.accounts TO pass_user;

GRANT ALL ON TABLE public.accounts TO postgres;

COMMENT ON TABLE public.accounts
    IS 'main table';
